# OSMOSIS.EXE Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single-file browser tool that blends two videos through luminance-gated pixel contamination into a persistent, drifting output buffer.

**Architecture:** Single `osmosis.html` — all HTML, CSS, JS inline. A persistent `ImageData` contamination buffer (854×480) is updated per frame by a pixel-level bleed algorithm and displayed on a CSS-scaled canvas. No build step, no external dependencies except Courier Prime (Google Fonts). Mirrors GLITCH.EXE's structure.

**Tech Stack:** HTML5 Canvas 2D, OffscreenCanvas, HTMLVideoElement, requestAnimationFrame, Uint8ClampedArray, MediaRecorder

```html
        .drop-zone.waiting { opacity: 0.45; }
        button:disabled { opacity: 0.3; cursor: not-allowed; }
    </style>
</head>
<body>
    <div id="app">
        <h1 style="font-size:20px;letter-spacing:0.15em;margin-bottom:24px;">OSMOSIS.EXE</h1>
        <div class="drop-zones">
            <div id="dropA" class="drop-zone waiting">[ DROP VIDEO A ]</div>
            <div id="dropB" class="drop-zone waiting">[ DROP VIDEO B ]</div>
        </div>
        <button id="playBtn" disabled>PLAY</button>
    </div>

    <input type="file" id="fileA" accept="video/*" style="display:none">
    <input type="file" id="fileB" accept="video/*" style="display:none">
    <video id="videoA" style="display:none" loop crossorigin="anonymous"></video>
    <video id="videoB" style="display:none" loop crossorigin="anonymous"></video>

    <script>
        const videoA = document.getElementById('videoA');
        const videoB = document.getElementById('videoB');
        const playBtn = document.getElementById('playBtn');

        let readyA = false, readyB = false;
        let isPlaying = false;

        function checkReady() {
            if (readyA && readyB) playBtn.disabled = false;
        }

        function loadVideo(video, file, zone, onReady) {
            zone.textContent = 'LOADING...';
            video.src = URL.createObjectURL(file);
            video.oncanplay = () => {
                const label = zone.id === 'dropA' ? '[ VIDEO A — READY ]' : '[ VIDEO B — READY ]';
                zone.textContent = label;
                zone.classList.remove('waiting');
                onReady();
            };
            video.load();
        }

        function setupZone(zoneId, fileInputId, video, onReady) {
            const zone = document.getElementById(zoneId);
            const input = document.getElementById(fileInputId);
            zone.addEventListener('click', () => input.click());
            input.addEventListener('change', e => {
                if (e.target.files[0]) loadVideo(video, e.target.files[0], zone, onReady);
            });
            zone.addEventListener('dragover', e => e.preventDefault());
            zone.addEventListener('drop', e => {
                e.preventDefault();
                const file = e.dataTransfer.files[0];
                if (file && file.type.startsWith('video/')) loadVideo(video, file, zone, onReady);
            });
        }

        setupZone('dropA', 'fileA', videoA, () => { readyA = true; checkReady(); });
        setupZone('dropB', 'fileB', videoB, () => { readyB = true; checkReady(); });

        playBtn.addEventListener('click', () => {
            if (!isPlaying) {
                videoA.play();
                videoB.play();
                isPlaying = true;
                playBtn.textContent = 'STOP';
            } else {
                videoA.pause();
                videoB.pause();
                isPlaying = false;
                playBtn.textContent = 'PLAY';
            }
        });

        // Freeze shorter video at its last frame instead of looping
        videoA.removeAttribute('loop');
        videoB.removeAttribute('loop');
        videoA.addEventListener('ended', () => videoA.pause());
        videoB.addEventListener('ended', () => videoB.pause());
    </script>
</body>
</html>
```

Note: `loop` attribute is removed in JS after the element is declared — the HTML has it for fallback, JS overrides it so shorter videos freeze at end rather than loop.

- [ ] **Step 2: Add bleedPixels function**

Add before `setupZone`:

```javascript
function bleedPixels(dataA, dataB, buf) {
    for (let i = 0; i < INTERNAL_W * INTERNAL_H; i++) {
        const r = i * 4, g = r + 1, b = r + 2;
        const lumA = 0.299 * dataA[r] + 0.587 * dataA[g] + 0.114 * dataA[b];
        const lumB = 0.299 * dataB[r] + 0.587 * dataB[g] + 0.114 * dataB[b];

        if ((direction === 'A_TO_B' || direction === 'BIDIRECTIONAL') && lumA > lumB) {
            const t = bleedRate * ((lumA - lumB) / 255);
            buf[r] += (dataA[r] - buf[r]) * t;
            buf[g] += (dataA[g] - buf[g]) * t;
            buf[b] += (dataA[b] - buf[b]) * t;
        }
        if ((direction === 'B_TO_A' || direction === 'BIDIRECTIONAL') && lumB > lumA) {
            const t = bleedRate * ((lumB - lumA) / 255);
            buf[r] += (dataB[r] - buf[r]) * t;
            buf[g] += (dataB[g] - buf[g]) * t;
            buf[b] += (dataB[b] - buf[b]) * t;
        }
    }
}
```

- [ ] **Step 3: Add render loop**

Add after `bleedPixels`:

```javascript
function renderFrame(timestamp) {
    requestAnimationFrame(renderFrame);
    if (!isPlaying || !bufInitialized) return;

    if (lastFrameTime > 0) {
        const delta = timestamp - lastFrameTime;
        currentFps = Math.round(1000 / delta);
        frameSkipActive = currentFps < 25;
    }
    lastFrameTime = timestamp;

    ctxA.drawImage(videoA, 0, 0, INTERNAL_W, INTERNAL_H);
    ctxB.drawImage(videoB, 0, 0, INTERNAL_W, INTERNAL_H);

    const dataA = ctxA.getImageData(0, 0, INTERNAL_W, INTERNAL_H).data;
    const dataB = ctxB.getImageData(0, 0, INTERNAL_W, INTERNAL_H).data;
    const buf = contaminationBuffer.data;

    if (frameSkipActive) {
        frameSkipToggle = !frameSkipToggle;
        if (!frameSkipToggle) bleedPixels(dataA, dataB, buf);
    } else {
        bleedPixels(dataA, dataB, buf);
    }

    displayCtx.putImageData(contaminationBuffer, 0, 0);
}

requestAnimationFrame(renderFrame);
```

```javascript
function computeContaminationPercent(threshold = 30) {
    if (!bufInitialized || !originalSnapshot) return 0;
    const buf = contaminationBuffer.data;
    let altered = 0;
    for (let i = 0; i < 1000; i++) {
        const idx = Math.floor(Math.random() * (INTERNAL_W * INTERNAL_H)) * 4;
        const dr = Math.abs(buf[idx]   - originalSnapshot[idx]);
        const dg = Math.abs(buf[idx+1] - originalSnapshot[idx+1]);
        const db = Math.abs(buf[idx+2] - originalSnapshot[idx+2]);
        if (dr + dg + db > threshold) altered++;
    }
    return altered / 1000;
}
```

```javascript
function startRecording() {
    const mimeType = MediaRecorder.isTypeSupported('video/webm;codecs=vp9')
        ? 'video/webm;codecs=vp9'
        : 'video/webm';
    const stream = displayCanvas.captureStream(30);
    recordedChunks = [];
    mediaRecorder = new MediaRecorder(stream, { mimeType });
    mediaRecorder.ondataavailable = e => {
        if (e.data.size > 0) recordedChunks.push(e.data);
    };
    mediaRecorder.onstop = () => {
        const blob = new Blob(recordedChunks, { type: 'video/webm' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = 'osmosis.webm';
        a.click();
        URL.revokeObjectURL(url);
    };
    mediaRecorder.start();
    isRecording = true;
    recBtn.textContent = 'STOP REC';
    recBtn.classList.add('active');
}

function stopRecording() {
    if (mediaRecorder && mediaRecorder.state !== 'inactive') mediaRecorder.stop();
    isRecording = false;
    recBtn.textContent = 'REC';
    recBtn.classList.remove('active');
}

recBtn.addEventListener('click', () => {
    if (!isRecording) startRecording();
    else stopRecording();
});
```
